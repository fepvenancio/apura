   ```typescript
   import Stripe from 'stripe';
   import type { PlanType } from '@apura/shared';
   import type { Env } from '../types';

   export function getStripe(env: Env): Stripe {
     return new Stripe(env.STRIPE_SECRET_KEY, {
       httpClient: Stripe.createFetchHttpClient(),
     });
   }

   // Map Stripe price IDs to internal plan types.
   // Price IDs are configured as env vars so they differ between test/prod.
   export function buildPriceMap(env: Env): Record<string, PlanType> {
     const map: Record<string, PlanType> = {};
     const prices: Array<[string | undefined, PlanType]> = [
       [(env as any).STRIPE_PRICE_STARTER_MONTHLY, 'starter'],
       [(env as any).STRIPE_PRICE_STARTER_ANNUAL, 'starter'],
       [(env as any).STRIPE_PRICE_PROFESSIONAL_MONTHLY, 'professional'],
       [(env as any).STRIPE_PRICE_PROFESSIONAL_ANNUAL, 'professional'],
       [(env as any).STRIPE_PRICE_BUSINESS_MONTHLY, 'business'],
       [(env as any).STRIPE_PRICE_BUSINESS_ANNUAL, 'business'],
       [(env as any).STRIPE_PRICE_ENTERPRISE_MONTHLY, 'enterprise'],
       [(env as any).STRIPE_PRICE_ENTERPRISE_ANNUAL, 'enterprise'],
     ];
     for (const [priceId, plan] of prices) {
       if (priceId) map[priceId] = plan;
     }
     return map;
   }

   export function getPlanFromPriceId(priceMap: Record<string, PlanType>, priceId: string): PlanType {
     return priceMap[priceId] ?? 'trial';
   }
   ```

1. Create `packages/api-gateway/src/routes/webhooks.ts`:
   - POST `/stripe` route
   - Read raw body with `await c.req.text()` FIRST (before any JSON parsing)
   - Verify signature with `stripe.webhooks.constructEventAsync(body, signature, env.STRIPE_WEBHOOK_SECRET, undefined, Stripe.createSubtleCryptoProvider())`
   - Return 400 if missing signature or invalid signature
   - Check idempotency: SELECT from stripe_events WHERE event_id = ?. If exists, return 200 `{ received: true }`.
   - Handle 6 event types in a switch:
     a. `checkout.session.completed`: Extract orgId from `session.client_reference_id`, customerId from `session.customer`, subscriptionId from `session.subscription`. Call `stripe.subscriptions.retrieve(subscriptionId)` to get price/plan. Use `getPlanFromPriceId()` to map. Call `orgDb.updateOrg({ stripe_customer_id, stripe_subscription_id, plan, max_users: limits.maxUsers, max_queries_per_month: limits.maxQueries, subscription_status: 'active', current_period_end })`.
     b. `customer.subscription.updated`: Extract orgId from `subscription.metadata.org_id`. Get price from `subscription.items.data[0]?.price.id`. Map to plan. Set `subscription_status` to `'canceling'` if `subscription.cancel_at_period_end` is true, otherwise use `subscription.status`. Update org with plan, limits, subscription_status, current_period_end.
     c. `customer.subscription.deleted`: Extract orgId from `subscription.metadata.org_id`. Downgrade to trial: `plan: 'trial'`, `max_queries_per_month: PLAN_LIMITS.trial.maxQueries`, `max_users: PLAN_LIMITS.trial.maxUsers`, `subscription_status: 'canceled'`, `stripe_subscription_id: null`.
     d. `invoice.payment_succeeded`: Extract subscription from `invoice.subscription`. Look up org by stripe_subscription_id. Set `subscription_status: 'active'`.
     e. `invoice.payment_failed`: Extract subscription from `invoice.subscription`. Look up org by stripe_subscription_id. Set `subscription_status: 'past_due'`. Use `c.executionCtx.waitUntil()` to enqueue a payment failure notification email via EMAIL_QUEUE: `{ type: 'payment_failed', to: [org.billing_email] }`.
     f. `invoice.payment_action_required`: Same as payment_failed -- set `subscription_status: 'past_due'`.
   - After processing, INSERT into stripe_events table.
   - Return `{ received: true }` with status 200.
   - For org lookup by stripe_subscription_id (invoice events), query DB directly: `SELECT id FROM organizations WHERE stripe_subscription_id = ?`.
