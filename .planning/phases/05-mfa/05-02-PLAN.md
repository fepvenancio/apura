<!-- Backend API contracts from Plan 01 -->

POST /api/mfa/setup (authenticated):
  Response: { success: true, data: { qrCodeDataUrl: string, secret: string } }
  Error: { success: false, error: { code: 'MFA_ALREADY_ENABLED', message: string } }

POST /api/mfa/confirm (authenticated):
  Body: { code: string }
  Response: { success: true, data: { backupCodes: string[] } }
  Error: { success: false, error: { code: 'INVALID_CODE' | 'NO_SETUP', message: string } }

POST /api/mfa/disable (authenticated):
  Body: { code: string }
  Response: { success: true }

DELETE /api/mfa/reset/:userId (authenticated, admin/owner):
  Response: { success: true }
  Error: { success: false, error: { code: 'FORBIDDEN', message: string } }

POST /auth/login:
  When MFA enabled: { success: true, data: { mfaRequired: true, mfaToken: string } }
  When MFA not enabled: { success: true, data: { accessToken, refreshToken, ... } }

POST /auth/mfa/verify:
  Body: { mfaToken: string, code: string }
  Response (success): { success: true, data: { accessToken, refreshToken, expiresIn, user } }
  Error: { success: false, error: { code: 'INVALID_CODE' | 'MFA_LOCKED' | 'INVALID_TOKEN', message: string } }
