Replace `.github/workflows/ci.yml` with a consolidated CI workflow:

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  ci:
    name: Lint, Typecheck & Test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Lint
        run: npm run lint

      - name: Typecheck
        run: npm run typecheck

      - name: Test
        run: npm run test

      - name: Build frontend
        run: npm run build --workspace=frontend

  preview:
    name: Preview Deploy
    needs: ci
    if: github.event_name == 'pull_request'
    runs-on: ubuntu-latest
    permissions:
      pull-requests: write
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build frontend
        working-directory: frontend
        run: npm run build

      - name: Deploy preview
        id: deploy
        working-directory: frontend
        run: |
          DEPLOY_OUTPUT=$(npx wrangler pages deploy out --project-name apura-web --branch "${{ github.head_ref }}" --commit-dirty=true 2>&1)
          echo "$DEPLOY_OUTPUT"
          PREVIEW_URL=$(echo "$DEPLOY_OUTPUT" | grep -oP 'https://[^\s]+\.apura-web\.pages\.dev' | head -1)
          echo "preview_url=$PREVIEW_URL" >> "$GITHUB_OUTPUT"
        env:
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}

      - name: Comment preview URL
        if: steps.deploy.outputs.preview_url
        uses: actions/github-script@v7
        with:
          script: |
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: `Preview deployed: ${{ steps.deploy.outputs.preview_url }}`
            })
```

Updated deploy.yml structure:

```yaml
name: Deploy

on:
  push:
    branches: [main]

concurrency:
  group: deploy-${{ github.ref }}
  cancel-in-progress: true

env:
  CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
  CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}

jobs:
  migrate:
    name: Run D1 Migrations
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run D1 migrations
        run: bash deploy/migrate.sh main

  deploy-workers:
    name: Deploy Workers
    needs: migrate
    runs-on: ubuntu-latest
    strategy:
      max-parallel: 1
      matrix:
        worker:
          - name: apura-ai
            path: packages/ai-orchestrator
          - name: apura-ws
            path: packages/ws-gateway
          - name: apura-api
            path: packages/api-gateway
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Deploy ${{ matrix.worker.name }}
        working-directory: ${{ matrix.worker.path }}
        run: npx wrangler deploy

  deploy-frontend:
    name: Deploy Frontend (Production)
    needs: migrate
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build frontend
        working-directory: frontend
        run: npm run build

      - name: Deploy to Cloudflare Pages
        working-directory: frontend
        run: npx wrangler pages deploy out --project-name apura-web --branch main --commit-dirty=true
```

Key changes:
1. Added `migrate` job that runs `deploy/migrate.sh main` before deployments
2. `deploy-workers` now `needs: migrate` (runs after migrations complete)
3. Uses `npm ci` at root instead of per-workspace `npm install`
4. Removed duplicate `test` job (CI workflow handles testing)
5. `deploy-frontend` uses `--branch main` for production deployment
