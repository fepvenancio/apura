   The existing language selector initializes to "pt" hardcoded. Wire it to:
   - Initialize from `user?.language || "pt"` (the AuthUser now has a language field from Plan 01)
   - On language change:
     a. Call `api.updateProfile({ language: newLang })` to persist to backend
     b. Update the auth store user object in localStorage with new language
     c. Navigate to `/${newLang}/settings/profile` to switch the app locale
   - The language change should immediately re-render the page in the new language because the URL locale segment changes, triggering NextIntlClientProvider to load the new message file.

   Updated code pattern:
   ```typescript
   const locale = useLocale();
   const [language, setLanguage] = useState(user?.language || locale);

   const handleProfileSubmit = async (e: React.FormEvent) => {
     e.preventDefault();
     setSaving(true);
     try {
       await api.updateProfile({ name, language });
       // Update user in localStorage with new language
       if (typeof window !== "undefined") {
         const stored = localStorage.getItem("user");
         if (stored) {
           const userData = JSON.parse(stored);
           userData.language = language;
           localStorage.setItem("user", JSON.stringify(userData));
         }
       }
       setProfileSuccess(true);
       // If language changed, navigate to new locale
       if (language !== locale) {
         router.push(`/${language}/settings/profile`);
         return;
       }
       setTimeout(() => setProfileSuccess(false), 3000);
     } catch {
       setProfileError(t("updateError"));
     } finally {
       setSaving(false);
     }
   };
   ```
